# ide.py - LogicStart IDE Backend FINAL from flask import Flask, request, jsonify from IA import IA from pymongo import MongoClient from datetime import datetime import os app = Flask(__name__) # ================= CONFIG ================= MONGO_URI = os.getenv("MONGO_URI", "mongodb://localhost:27017/") API_KEY = os.getenv("OPENAI_API_KEY") # ================= INIT ================= ia = IA(api_key=API_KEY) mongo = MongoClient(MONGO_URI) db = mongo["logicstart"] files_col = db["files"] logs_col = db["logs"] # ================= UTILS ================= def ok(data=None): return jsonify({ "status": "ok", "data": data }) def erro(msg, code=400): return jsonify({ "status": "erro", "mensagem": msg }), code def log_event(user_id, acao, detalhe=""): logs_col.insert_one({ "user_id": user_id, "acao": acao, "detalhe": detalhe, "timestamp": datetime.utcnow() }) # ================= HEALTH ================= @app.route("/") def home(): return ok("🚀 LogicStart IDE ONLINE") # ================= ARQUIVOS ================= @app.route("/files", methods=["GET"]) def listar_arquivos(): user_id = request.args.get("user_id", "default") arquivos = list(files_col.find( {"user_id": user_id}, {"_id": 0} )) return ok(arquivos) @app.route("/files", methods=["POST"]) def salvar_arquivo(): data = request.json user_id = data.get("user_id", "default") nome = data.get("nome") codigo = data.get("codigo", "") if not nome: return erro("Nome obrigatório") files_col.update_one( {"user_id": user_id, "nome": nome}, { "$set": { "codigo": codigo, "updated_at": datetime.utcnow() } }, upsert=True ) # 🔥 salva no cérebro da IA ia.salvar_codigo(user_id, nome, codigo) log_event(user_id, "salvar_arquivo", nome) return ok("Arquivo salvo") @app.route("/files/", methods=["GET"]) def abrir_arquivo(nome): user_id = request.args.get("user_id", "default") arq = files_col.find_one( {"user_id": user_id, "nome": nome}, {"_id": 0} ) if not arq: return erro("Arquivo não encontrado", 404) return ok(arq) @app.route("/files/", methods=["DELETE"]) def deletar_arquivo(nome): user_id = request.args.get("user_id", "default") files_col.delete_one({ "user_id": user_id, "nome": nome }) log_event(user_id, "deletar_arquivo", nome) return ok("Arquivo deletado") # ================= IA ================= @app.route("/ia/chat", methods=["POST"]) def chat(): data = request.json user_id = data.get("user_id", "default") mensagem = data.get("mensagem", "") resposta = ia.perguntar(user_id, mensagem) log_event(user_id, "chat", mensagem) return ok(resposta) @app.route("/ia/gerar", methods=["POST"]) def gerar(): data = request.json user_id = data.get("user_id", "default") pedido = data.get("pedido", "") codigo = ia.gerar(user_id, pedido) log_event(user_id, "gerar_codigo", pedido) return ok(codigo) @app.route("/ia/corrigir", methods=["POST"]) def corrigir(): data = request.json user_id = data.get("user_id", "default") codigo = data.get("codigo", "") novo = ia.corrigir(user_id, codigo) log_event(user_id, "corrigir_codigo") return ok(novo) @app.route("/ia/explicar", methods=["POST"]) def explicar(): data = request.json user_id = data.get("user_id", "default") codigo = data.get("codigo", "") texto = ia.explicar(user_id, codigo) return ok(texto) @app.route("/ia/sugerir", methods=["POST"]) def sugerir(): data = request.json codigo = data.get("codigo", "") sugestoes = ia.sugerir(codigo) return ok(sugestoes) @app.route("/ia/analisar", methods=["POST"]) def analisar(): data = request.json codigo = data.get("codigo", "") erros = ia.analisar(codigo) return ok(erros) # ================= EXECUÇÃO ================= @app.route("/executar", methods=["POST"]) def executar(): data = request.json codigo = data.get("codigo", "") user_id = data.get("user_id", "default") try: from engine import Engine engine = Engine(codigo) engine.executar() log_event(user_id, "executar_codigo") return ok({ "saida": engine.saida }) except Exception as e: return erro(str(e), 500) # ================= LOGS ================= @app.route("/logs", methods=["GET"]) def logs(): user_id = request.args.get("user_id", "default") registros = list( logs_col.find({"user_id": user_id}, {"_id": 0}) .sort("timestamp", -1) .limit(50) ) return ok(registros) # ================= SEGURANÇA BÁSICA ================= @app.before_request def validar_json(): if request.method == "POST": if not request.is_json: return erro("Requisição deve ser JSON", 415) # ================= RUN ================= if __name__ == "__main__": print("🚀 LogicStart IDE rodando em produção...") app.run(host="0.0.0.0", port=5000)
